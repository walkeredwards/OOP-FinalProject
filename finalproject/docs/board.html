304    def move(self, color: str, selected_piece: Piece | King,
305             new_location: tuple[int, int]) -> bool:
306        """moves a piece
307
308        Args:
309            color (str): color of player
310            selected_piece : selected piece fo player
311            new_location (tuple): location waning to move piece to
312
313        Returns:
314            bool: indicates if piece moved or not
315        """
316        # setup
317        move = False
318        last_location: tuple[int, int]
319
320        # adds enpassant pawn to piece and location list
321        if self._enpassant_pawn is not None and isinstance(selected_piece, Pawn):
322            if color == "white":
323                self._black_pieces.append(self._enpassant_pawn)
324                self._black_location.append(self._enpassant_pawn.location)
325            else:
326                self._white_pieces.append(self._enpassant_pawn)
327                self._white_location.append(self._enpassant_pawn.location)
328
329        # find actual moves for piece
330        actual_moves = self.actual_moves(color, selected_piece)
331
332        # checks if available and moves if new_location is in actual_moves
333        if new_location in actual_moves:
334            if isinstance(selected_piece, King):
335                last_location = selected_piece.location
336                move = selected_piece.move(new_location, self._black_location, self._white_location,
337                                           self._black_pieces, self._white_pieces)
338            else:
339                last_location = selected_piece.location
340                move = selected_piece.move(new_location, self._black_location, self._white_location)
341
342        # removes enpassant pawn from piece list and location list
343        if self._enpassant_pawn is not None and isinstance(selected_piece, Pawn):
344            if color == "white":
345                self._black_pieces.remove(self._enpassant_pawn)
346                self._black_location.remove(self._enpassant_pawn.location)
347            else:
348                self._white_pieces.remove(self._enpassant_pawn)
349                self._white_location.remove(self._enpassant_pawn.location)
350
351        # if the piece actually moved update info
352        if move:
353            if self._enpassant_pawn is not None and self._last_piece_moved is not None and \
354                    new_location == self._enpassant_pawn.location:
355                # captures the enpassant pawn
356                self.check_capture(color, self._last_piece_moved.location)
357
358            self._old_location = last_location
359            self._last_piece_moved = selected_piece
360
361            if isinstance(selected_piece, Pawn):
362                if (selected_piece.color == 'black' and selected_piece.location[1] == 7):
363                    self.promotion(selected_piece)
364                elif (selected_piece.color == 'white' and selected_piece.location[1] == 0):
365                    self.promotion(selected_piece)
366
367            # if a pawn moved 2 tile set up for enpassant
368            if isinstance(selected_piece, Pawn) and \
369                    (new_location[1] == self._old_location[1] + 2 or
370                     new_location[1] == self._old_location[1] - 2):
371                self.enpassant(selected_piece, True)
372            else:
373                # reset enpassant to false
374                self.enpassant(selected_piece, False)
375
376            # captures any pieces
377            self.check_capture(color, new_location)
378            # update piece locations
379            self.update_locations()
380
381        return move

550        """finds the tiles that can block a check
551
552        Args:

554
555        Returns:
556            list: list of blockable spaces
557        """
558        attacking = self._last_piece_moved  # piece that is attacking king
559        block_tile: list[tuple[int, int]] = []
560        distance = 1  # start at a tile away
561        temp_coord = (0, 0)
562        direction = (0, 0)
563
564        if attacking is not None:
565            if turn == "white":
566                king = self._white_king
567            else:
568                king = self._black_king
569
570            # no blocking knight
571            if not isinstance(attacking, Knight):
572                # finds if king is to right or left of attacker
573                x = king.location[0] - attacking.location[0]

580                elif y == 0:
581                    direction = (x // abs(x), y)
582                else:
583                    direction = (x // abs(x), y // abs(y))
584
585                # goes from piece to king location appending blockable tiles
586                while temp_coord != king.location:
587                    temp_coord = (attacking.location[0] + direction[0] * distance,
588                                  attacking.location[1] + direction[1] * distance)
589                    block_tile.append(temp_coord)
590                    distance += 1
591
592            # adds attacking location for capture
593            block_tile.append(attacking.location)
594
595        return block_tile

597    def actual_moves(self, color: str, piece: Piece | King) -> list[tuple[int, int]]:
598        """finds the actual available moves for a piece

600        Args:
601            color (str): color of piece
602            piece (Piece): piece
603
604        Returns:
605            list: list of actual valid moves.
606        """
607        moves = []
608
609        if isinstance(piece, King):
610            # finds king's possible moves
611            moves = piece.possible_moves(self._black_location, self._white_location,
612                                         self._black_pieces, self._white_pieces, color)
613            # can't castle while king is in check
614            if self._in_check is not None and self._in_check.color == color:
615                if color == "white" and (2, 7) in moves:
616                    moves.remove((2, 7))
617                if color == "white" and (6, 7) in moves:
618                    moves.remove((6, 7))
619                if color == "black" and (2, 0) in moves:
620                    moves.remove((2, 0))
621                if color == "black" and (6, 0) in moves:
622                    moves.remove((6, 0))
623
624        else:
625            # finds possible moves for the piece
626            moves = piece.possible_moves(self._black_location, self._white_location)
627
628            # check block handeling, king can't block only move out
629            if self._in_check is not None:  # if a king is in check
630                selected_moves = moves  # moves list of moves to append
631                moves = []  # clears moves
632                block_moves = self.in_check_block(color)  # finds blockable tiles
633                for coord in selected_moves:
634                    if coord in block_moves:
635                        # if the possible move is in blockable moves append to available
636                        moves.append(coord)
637
638            # check if move dosn't put king in check
639            moves = self.check_can_move(color, piece, moves)

644                       moves: list[tuple[int, int]]) -> list[tuple[int, int]]:

649            piece (_type_): piece to be checked
650            moves (list): list of possible moves
651
652        Returns:
653            list: list of moves that won't put king in check

656        if color == "white":

669
670        tmp_enemy = None
671        check_move = moves
672        moves = []
673        safe = True
674        # removes pece and piece location from lists
675        temp_location.remove(piece.location)
676        temp_piece.remove(piece)
677
678        for coord in check_move:

681            # adds possible location
682            temp_location.append(coord)
683            temp_piece.append(piece_type)
684            capture_check = False
685            # adds capture
686            for enemy_piece in enemies:

688                    tmp_enemy = enemy_piece
689                    capture_check = True
690                    tmp_enemy_location.remove(tmp_enemy.location)
691                    tmp_enemies.remove(tmp_enemy)

700                                                temp_piece, tmp_enemies,

703            if safe:
704                moves.append(coord)
705            # removes new pece infomation

720
721        Args:

723
724        Returns:
725            bool: if king is in checkmate
726        """
727        king_can_move = False
728        can_block = False
729        block_tiles = self.in_check_block(color)
730
731        # sets friends
732        if color == "white":
733            friend = self._white_pieces
734        else:
735            friend = self._black_pieces
736
737        for piece in friend:
738            if not (can_block or king_can_move):

741                                                 self._black_pieces, self._white_pieces, color)

743                    if len(moves) > 0:
744                        return False
745                else:
746                    moves = piece.possible_moves(self._black_location, self._white_location)
747                    if not (can_block or king_can_move):
748                        for coord in moves:
749                            # if piece can block check
750                            if coord in block_tiles:
751                                return False

757    def check_stalemate(self, color: str) -> bool:
758        """checks for stalemate
759        Will only recognize if king not in check and no piece can move
760        Won't recongize unwinnable states
761        ex: w_king w_rook and b_king, w_king w_knight and b_king, etc
762
763        Args:
764            color (str): color that just moved
765
766        Returns:
767            bool: is game state a stalemate
768        """
769        # sets friends
770        if color == "white":
771            friend = self._black_pieces
772            other_color = "black"
773        else:
774            friend = self._white_pieces
775            other_color = "white"
776
777        for piece in friend:
778            if isinstance(piece, King):
779                moves = piece.possible_moves(self._black_location, self._white_location,
780                                             self._black_pieces, self._white_pieces, color)
781                # if king can move return false
782                if len(moves) > 0:
783                    return False
784
785            else:
786                moves = self.actual_moves(other_color, piece)
787                # if any piece can move
788                if len(moves) > 0:
789                    return False
790
791        # there is no check
792        if self._in_check is None:
793            stalemate = True
794        else:
795            stalemate = False
796
797        # if only one piece remains
798        if len(self._white_pieces) == 1 and len(self._black_pieces) == 1:
799            stalemate = True
800
801        return stalemate

827        if checkmate:
828            self.end_game(color)

832            return True

550    def in_check_block(self, turn: str) -> list[tuple[int, int]]:
551        """finds the tiles that can block a check
552
553        Args:
554            turn (str): color of the person that can move
555
556        Returns:
557            list: list of blockable spaces
558        """
559        attacking = self._last_piece_moved  # piece that is attacking king
560        block_tile: list[tuple[int, int]] = []
561        distance = 1  # start at a tile away
562        temp_coord = (0, 0)
563        direction = (0, 0)
564
565        if attacking is not None:
566            if turn == "white":
567                king = self._white_king
568            else:
569                king = self._black_king
570
571            # no blocking knight
572            if not isinstance(attacking, Knight):
573                # finds if king is to right or left of attacker
574                x = king.location[0] - attacking.location[0]
575                # finds if king is up or down of attacker
576                y = king.location[1] - attacking.location[1]
577
578                # calculates direction of attaker to king,
579                if x == 0:
580                    direction = (x, y // abs(y))
581                elif y == 0:
582                    direction = (x // abs(x), y)
583                else:
584                    direction = (x // abs(x), y // abs(y))
585
586                # goes from piece to king location appending blockable tiles
587                while temp_coord != king.location:
588                    temp_coord = (attacking.location[0] + direction[0] * distance,
589                                  attacking.location[1] + direction[1] * distance)
590                    block_tile.append(temp_coord)
591                    distance += 1
592
593            # adds attacking location for capture
594            block_tile.append(attacking.location)
595
596        return block_tile

758    def check_stalemate(self, color: str) -> bool:
759        """checks for stalemate
760        Will only recognize if king not in check and no piece can move
761        Won't recongize unwinnable states
762        ex: w_king w_rook and b_king, w_king w_knight and b_king, etc
763
764        Args:
765            color (str): color that just moved
766
767        Returns:
768            bool: is game state a stalemate
769        """
770        # sets friends
771        if color == "white":
772            friend = self._black_pieces
773            other_color = "black"
774        else:
775            friend = self._white_pieces
776            other_color = "white"
777
778        for piece in friend:
779            if isinstance(piece, King):
780                moves = piece.possible_moves(self._black_location, self._white_location,
781                                             self._black_pieces, self._white_pieces, color)
782                # if king can move return false
783                if len(moves) > 0:
784                    return False
785
786            else:
787                moves = self.actual_moves(other_color, piece)
788                # if any piece can move
789                if len(moves) > 0:
790                    return False
791
792        # there is no check
793        if self._in_check is None:
794            stalemate = True
795        else:
796            stalemate = False
797
798        # if only one piece remains
799        if len(self._white_pieces) == 1 and len(self._black_pieces) == 1:
800            stalemate = True
801
802        return stalemate
